Effects

Opt in to scale, fade and / or motionBlur effects. Opt out of horizontal slide, or change the amount of horizontal sliding.

TinSlide(document.getElementById('slider'), {
  effects: {
    slideHorizontal: {
      offset: 0.3
    },
    scale: {
      on: true
    },
    fade: {
      on: true
    },
    motionBlur: {
      on: true
    }
  },
  cropContainer: false
});
