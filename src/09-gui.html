GUI

Opt out of non touch swipe navigation, and dots navigation.

TinSlide(document.getElementById('slider'), {
  useSwipeNavigation: true, // Default
  swipeTouchOnly: true,
  generate: {
    dots: {
      on: false
    },
    nav: {
      on: true // Default
    }
  }
});
